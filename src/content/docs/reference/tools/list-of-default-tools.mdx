import { getCollection } from 'astro:content';

export const EXCLUDED_TOOL_PAGE_IDS = [
  'reference/tools/introduction',
  'reference/tools/list-of-default-tools',
];

export async function getToolsList() {
  const tools = await getCollection('docs', (entry) => {
    if (!entry.id) return false;
    return entry.id.startsWith('reference/tools') && !EXCLUDED_TOOL_PAGE_IDS.includes(entry.id);
  });
  
  return tools.sort((a, b) => a.data.title.localeCompare(b.data.title));
}

export const tools = await getToolsList();

The complete list of Alation-provided tools is available in your Agent Studio Tools page `https://your-instance.alationcloud.com/app/studio/tools`.
