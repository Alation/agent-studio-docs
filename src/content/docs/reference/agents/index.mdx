export const EXCLUDED_AGENT_PAGE_IDS = [
  'reference/agents/introduction',
  'reference/agents',
];

import { getCollection } from 'astro:content';

export async function getAgentsList() {
  const agents = await getCollection('docs', (entry) => {
    if (!entry.id) return false;
    return entry.id.startsWith('reference/agents') && !EXCLUDED_AGENT_PAGE_IDS.includes(entry.id);
  });
  
  return agents.sort((a, b) => a.data.title.localeCompare(b.data.title));
}

export const agents = await getAgentsList();

The following agents are available out-of-the-box in Agent Studio:

<ul>
{agents.map(agent => (
  <li>
    <a href={`/agent-studio-docs/${agent.id}/`}>{agent.data.title}</a>
    {agent.data.description && ` - ${agent.data.description}`}
  </li>
))}
</ul>
